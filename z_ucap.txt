into @productno,@products,@hours,@minutes,@stationno,@stations,@tggno,@comp,@BBsCount

begin 
	declare @status nvarchar(max) = ''
	if(@BBsCount=0)

	fetch next from cursor_table 
	into @productno,@products,@hours,@minutes,@stationno,@stations,@tggno,@comp,@BBsCount
end
