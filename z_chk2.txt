z_chk2a:--z_chk2a
declare @t_noa nvarchar(30)
set @t_noa = case when '#non'=[1] then '' else [1] end
declare @tmp table(
	gno nvarchar(1),
	anoa nvarchar(30),
	adatea nvarchar(10),
	abankno nvarchar(20),
	abank nvarchar(90),
	aaccount nvarchar(30),
	amoney float,
	bcheckno nvarchar(30),
	bbankno nvarchar(20),
	bbank nvarchar(90),
	baccount nvarchar(30),
	ccomp nvarchar(90),
	bdatea nvarchar(10),
	bmoney float	
)
insert into @tmp
	select '0',a.noa,a.datea,a.bankno,a.bank,a.account,a.money,
		   b.checkno,b.bankno,b.bank,b.account,c.comp,b.datea,b.money
	from chk2 a
	left join chk2s b on a.noa = b.noa
	left join gqb c on b.checkno = c.gqbno
	where a.noa = @t_noa
	order by b.noq
select *,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,amoney),1)),4,12)) money1,
reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,bmoney),1)),4,12)) money2
from @tmp;
