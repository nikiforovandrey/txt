	IF OBJECT_ID('tempdb..#z_paytran1')is not null
	BEGIN
		set @cmd = 'drop table #z_paytran1'
		EXECUTE sp_executesql @cmd
	END
	create table #z_paytran1(
		gno nvarchar(1),
		driverno nvarchar(20),
		driver nvarchar(50),
		[money] float,
		pay float,
		unpay float,
		plusmoney float,
		minusmoney float,
		total float
	)
	
	set @cmd=
	" select '0',isnull(a.driverno,''),isnull(b.namea,''),"+
	" sum(ISNULL(a.total2,0)),SUM(ISNULL(c.[money],0)),0,0,0,0"+
	" from trans"+@t_accy+" a"+
	" left join driver b on a.driverno=b.noa"+ 
	" left join tres"+@t_accy+" c on a.noa=c.tranno and a.noq=c.trannoq"+
	" left join calctypes d  on a.calctype=d.noa+d.noq"+
	" where d.isoutside=1"+ 
	" and (a.driverno between @t_bdriverno and @t_edriverno)"+
	" and (a.datea between @t_bdate and @t_edate)"+
	" and (len(@t_rc2no)=0 or a.noa=@t_rc2no)"+
	" group  by  isnull(a.driverno,''),isnull(b.namea,'')"

	insert into #z_paytran1
	execute sp_executesql @cmd,N'@t_bdate nvarchar(10),@t_edate nvarchar(10),@t_bdriverno nvarchar(20),@t_edriverno nvarchar(20),@t_rc2no nvarchar(20)',
		@t_bdate=@t_bdate,@t_edate=@t_edate,@t_bdriverno=@t_bdriverno,@t_edriverno=@t_edriverno,@t_rc2no=@t_rc2no
	
	declare cursor_table cursor for
	select driverno,sum(isnull(plusmoney,0)),sum(isnull(minusmoney,0)) from carchg 
	where (datea between @t_bdate and @t_edate) and (driverno between @t_bdriverno and @t_edriverno)
	group by driverno
	open cursor_table
	fetch next from cursor_table
	into @driverno,@plusmoney,@minusmoney
	while(@@FETCH_STATUS <> -1)
	begin
		if not exists(select * from #z_paytran1 where driverno=@driverno)
		begin
			set @driver=''
			select @driver=namea from driver where noa=@driverno
			insert into #z_paytran1(gno,driverno,driver)values('0',@driverno,@driver)
		end
		update #z_paytran1 set plusmoney=@plusmoney,minusmoney=@minusmoney where driverno=@driverno
		fetch next from cursor_table
		into @driverno,@plusmoney,@minusmoney
	end
	close cursor_table
	deallocate cursor_table
	
	insert into #z_paytran1 select '1','','',SUM([money]),SUM(pay),0,SUM(plusmoney),SUM(minusmoney),0 from #z_paytran1
	update #z_paytran1 set unpay=[money]-pay,total=[money]+plusmoney-minusmoney
	
	select gno,driverno aa,driver bb
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,[money]),1)),4,12)) [money]
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,pay),1)),4,12)) pay
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,unpay),1)),4,12)) unpay
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,plusmoney),1)),4,12)) pm
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,minusmoney),1)),4,12)) mm
	,reverse(substring(reverse(convert(nvarchar(15),CONVERT(money,total),1)),4,12)) total
	from #z_paytran1 order by gno,driverno;
