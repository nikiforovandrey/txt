declare @t_result table(
		tggno nvarchar(10),
		tgg nvarchar(40),
		datea nvarchar(10),
		bccno nvarchar(20),
		bccname nvarchar(50),
		price float,
		memo nvarchar(200),
		beginprice float
)
insert into @t_result
select a.tggno,a.tgg,left(a.datea,6) datea,b.bccno,b.bccname,b.price,c.memo,c.price
from bccin a left join bccins b on a.noa=b.noa left join bcc c on b.bccno=c.noa
where a.datea between @t_year+'/01/01' and @t_year+'/12/31' and (tggno between @t_btggno and @t_etggno) and (b.bccno between @t_bbccno and @t_ebccno)
