	declare @recno int
	declare @productno nvarchar(30)
	declare @typea nvarchar(10)
	declare @datea nvarchar(10)
	declare @mount decimal(18,2)
	declare @weight decimal(18,2)
	
	declare @t_productno nvarchar(30)
	declare @t_mount decimal(18,2)
	declare @t_weight decimal(18,2)
	declare @t_totmount decimal(18,2)
	declare @t_totweight decimal(18,2)
	set @t_productno = '@#S(DJ#SH!@'
	set @t_mount = 0
	set @t_weight = 0
	set @t_totmount = 0
	set @t_totweight = 0
	
	declare cursor_table cursor for
	select recno,productno,typea,datea,mount,weight from @result order by productno,recno
	open cursor_table
	fetch next from cursor_table
	into @recno,@productno,@typea,@datea,@mount,@weight
	while(@@FETCH_STATUS <> -1)
	begin
		if @t_productno!=@productno and @t_productno!='@#S(DJ#SH!@'
		begin
			insert into @result
			select '1' gno,-1 recno,@t_productno,'' xproduct,'' datea,'zz' typea,'小計' typec, 
				   '' noa,'' comp, '' unit, @t_mount mount, @t_weight weight, @t_totmount totmount, @t_totweight totweight,''
		end
		if @t_productno!=@productno
		begin
			set @t_productno = @productno
			set @t_mount = 0
			set @t_weight = 0
			set @t_totmount = 0
			set @t_totweight = 0
		end
		if  @datea between @t_bdate and @t_edate
		begin
			set @t_mount =
				case 
				when @typea='01' or @typea='02' then @mount
				when @typea='10' or @typea='11' or @typea='12' then @t_mount + @mount
				when @typea='A0' or @typea='B0' or @typea='C0' then @t_mount - @mount
				end
			set @t_weight =
				case 
				when @typea='01' or @typea='02' then @weight
				when @typea='10' or @typea='11' or @typea='12' then @t_weight + @weight
				when @typea='A0' or @typea='B0' or @typea='C0' then @t_weight - @weight
				end
		end
		set @t_totmount =
			case 
			when @typea='01' or @typea='02' then @mount
			when @typea='10' or @typea='11' or @typea='12' then @t_totmount + @mount
			when @typea='A0' or @typea='B0' or @typea='C0' then @t_totmount - @mount
			end
		set @t_totweight =
			case 
			when @typea='01' or @typea='02' then @weight
			when @typea='10' or @typea='11' or @typea='12' then @t_totweight + @weight
			when @typea='A0' or @typea='B0' or @typea='C0' then @t_totweight - @weight
			end
		
		update @result set totmount = @t_totmount, totweight = @t_totweight where productno=@productno and recno=@recno

		fetch next from cursor_table
		into @recno,@productno,@typea,@datea,@mount,@weight
	end
	close cursor_table
	deallocate cursor_table
	if @t_productno!='@#S(DJ#SH!@'
	begin
		insert into @result
		select '1' gno,-1 recno,@t_productno,'' xproduct,'' datea,'zz' typea,'小計' typec, 
			   '' noa,'' comp, '' unit, @t_mount mount, @t_weight weight, @t_totmount totmount, @t_totweight totweigh,''
	end
	
	declare @ucctype nvarchar(15)
	declare cursor_table cursor for
	select noa,typea from ucc order by noa
	open cursor_table
	fetch next from cursor_table
	into @t_productno,@ucctype
	while(@@FETCH_STATUS <> -1)
	begin
		update @result set ucctype = @ucctype where productno=@t_productno
		fetch next from cursor_table
		into @t_productno,@ucctype
	end
	close cursor_table
	deallocate cursor_table
	
	declare cursor_table cursor for
	select noa,typea from uca order by noa
	open cursor_table
	fetch next from cursor_table
	into @t_productno,@ucctype
	while(@@FETCH_STATUS <> -1)
	begin
		update @result set ucctype = @ucctype where productno=@t_productno
		fetch next from cursor_table
		into @t_productno,@ucctype
	end
	close cursor_table
	deallocate cursor_table
	--*****************************************************************************************	
	select * from @result where (gno='1' or (gno='0' and datea between @t_bdate and @t_edate)) 
	 and (len(@t_typea) = 0 or ucctype = @t_typea)
	order by productno, gno, recno;

z_stk:	--庫存表  --z_stk
	declare @t_bproductno nvarchar(50)
	declare @t_eproductno nvarchar(50)
	declare @enddate nvarchar(15)
	declare @t_typea nvarchar(15)
	declare @t_bstoreno nvarchar(50) 
	declare @t_estoreno nvarchar(50)
	declare @t_groupano nvarchar(max)
	declare @t_style nvarchar(max)
	declare @allucc nvarchar(30)
	
	set @t_bproductno = case when '#non'=[4] then '' else [4] end
	set @t_eproductno = case when '#non'=[5] then char(255) else [5] end
	set @enddate=case when '#non'=[8] then '' else [8] end
	set @t_typea = case when '#non'=[11] then '' else [11] end
	set @t_bstoreno = case when '#non' = [6] then '' else [6] end
	set @t_estoreno = case when '#non' = [7] then CHAR(255) else [7] end
	set @t_groupano = case when '#non'=[13] then '' else [13] end
	set @t_style = case when '#non'=[14] then '' else [14] end
	set @allucc=case when '#non'=[15] then '0' else [15] end

	declare @t_stkdate nvarchar(30)
	set @t_stkdate=CONVERT (VARCHAR(7), GETDATE(),12 )+0890000
	set @t_stkdate=left(@t_stkdate,3)+'/'+substring(@t_stkdate,4,2)+'/'+right(@t_stkdate,2)
	
	--@t_typea=1@商品,2@製成品,3@半成品,4@原料,5@物料,6@下腳品,7@加工,8@零件,9@其它
	
	declare @result table(
		gno nvarchar(1),
		productno nvarchar(30),
		xproduct nvarchar(40),
		unit nvarchar(10),
		totmount float,
		price float,
		total float
	)
	

	if(len(@t_typea)=0)
	begin
		insert into @result
		select '0' gno,a.productno,a.product xproduct,a.unit,a.mount totmount,b.price,a.mount*b.price total 
		from (
			select s.productno,MAX(s.product) product,sum(s.mount) mount,(select top 1 unit from view_ucaucc where noa=s.productno) unit 
			from stkucc(@enddate,'','') s where (s.storeno between @t_bstoreno and @t_estoreno) group by productno
		) a
		left join (select * from costs[1] where mon=LEFT(@enddate,6)) b on a.productno=b.productno
		outer apply(select top 1 groupano,style from view_ucaucc where a.productno=noa) c
		where (a.productno!='') and (len(@t_groupano)=0 or @t_groupano=c.groupano)
		and a.productno between @t_bproductno and @t_eproductno
		and (len(@t_style)=0 or @t_style=c.style) 
	end 
	else
	begin
		if(@t_typea!='2' and @t_typea!='3')
		begin
			insert into @result
			select '0' gno,a.productno,a.product xproduct,b.unit,a.mount totmount,c.price,a.mount*c.price total
			from (
				select productno,MAX(product) product,sum(mount) mount 	from stkucc(@enddate,'','') s
				where (s.storeno between @t_bstoreno and @t_estoreno) group by productno
			) a
			left join ucc b on a.productno=b.noa  
			left join (select * from costs[1] where mon=LEFT(@enddate,6)) c on a.productno=c.productno
			outer apply(select top 1 groupano from view_ucaucc where a.productno=noa) d
			where (a.productno in (select noa from ucc)) and (b.typea=@t_typea) and (len(@t_groupano)=0 or @t_groupano=d.groupano)
			and a.productno between @t_bproductno and @t_eproductno and (len(@t_style)=0 or @t_style=b.style) 
		end
		else
		begin
			insert into @result
			select '0' gno,a.productno,a.product xproduct,b.unit,a.mount totmount,c.price,a.mount*c.price total
			from (
				select productno,MAX(product) product,sum(mount) mount from stkucc(@enddate,'','') s
				where (s.storeno between @t_bstoreno and @t_estoreno) group by productno
			) a
			left join (
				select noa,product,unit,typea from uca union all select productno,productno+' 半成品',''unit,'2' typea from ucat where productno not in (select noa from uca) and productno!=''
			) b on a.productno=b.noa 
			left join (select * from costs[1] where mon=LEFT(@enddate,6)) c on a.productno=c.productno
			outer apply(select top 1 groupano,style from view_ucaucc where a.productno=noa) d
			where (a.productno in (select noa from uca union select productno from ucat))
				  and (b.typea=@t_typea)
				  and (len(@t_groupano)=0 or @t_groupano=d.groupano)
				  and (a.productno between @t_bproductno and @t_eproductno)
				  and (len(@t_style)=0 or @t_style=d.style) 
		end
	end
	
	if(@allucc='0')
	begin
		delete @result where totmount=0
	end

	select gno,productno,xproduct,unit,@enddate enddate
		,reverse(substring(reverse(convert(nvarchar(30),CONVERT(money,round(totmount,0)),1)),4,30)) totmount
		,reverse(substring(reverse(convert(nvarchar(30),CONVERT(money,round(price,2)),1)),0,30)) price
		,reverse(substring(reverse(convert(nvarchar(30),CONVERT(money,round(total,0)),1)),4,30)) total
	from @result
;
